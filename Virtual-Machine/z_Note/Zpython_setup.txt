ADD PATH: Tùy trường hợp mà xem xét có nên bỏ kí tự $ cạnh HOME không ?

export PATH="$HOME/bin:$PATH"  // import path trong lần sử dụng hiện tại

import lưu luôn vô path trong những lần sau:

nano ~/.bashrc 

export PATH="$HOME/bin:$PATH"	// thêm vào cuối

source ~/.bashrc		// lưu vào

echo $PATH		 	// kiểm tra path
